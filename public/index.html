// script.google.com → New → วาง code:

function doPost(e) {
  var sh = SpreadsheetApp
    .getActiveSpreadsheet().getActiveSheet();
  var d = JSON.parse(e.postData.contents);
  if (sh.getLastRow()===0) sh.appendRow(
    ['วันที่','เวลา','ประเภท','จำนวน',
     'รายละเอียด','Tags']);
  var cm={income:'รายรับ',daily:'กิน/ช้อป',
    fixed:'ผ่อน/บัตร/ประกัน',saving:'ออม'};
  d.transactions.forEach(function(t){
    sh.appendRow([t.date,t.time,cm[t.type],
      t.amount,t.note,(t.tags||[]).join(', ')]);
  });
  return ContentService.createTextOutput('OK')
    .setMimeType(ContentService.MimeType.TEXT);
}
